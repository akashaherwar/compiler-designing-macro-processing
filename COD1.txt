* MACRO * *
* PQR &A1,&A2,&A3 *
* MOVER AREG &A1
* MOVEM BREG &A2
* MOVE CREG &A3
* MEND * *
* MACRO * *
&L XYZ &A1,&A2,&A3 *
* MOVER AREG &A1
&L MOVEM BREG &A2
* MOVE CREG &A3
* MEND * *
* MACRO * *
* ABC &ARG1=DATA3,&ARG2=DATA1,&ARG3=DATA2 *
* ADD AREG &ARG1
* ADD BREG &ARG2
* SUB BREG &ARG3
* MEND * *
* START * *
* PQR A,B,C *
LOOP XYZ X,Y,Z *
* ABC &ARG3=DATA2,&ARG2=DATA1 *
* END * *
